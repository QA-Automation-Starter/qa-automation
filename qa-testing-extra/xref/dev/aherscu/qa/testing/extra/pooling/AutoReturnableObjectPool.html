17  package dev.aherscu.qa.testing.extra.pooling;
18  
19  import java.util.*;
20  import java.util.concurrent.*;
21  
22  import org.apache.commons.pool2.impl.*;
23  
24  import lombok.*;
25  import lombok.extern.slf4j.*;
26  
27  /**
28   * Pool of object to be shared across parallel running tests. As long as a
29   * pooled object is borrowed it will not be borrowed to other running test.
30   * Borrowed pooled objects shall be returned as soon as possible in order not
31   * block other tests running in parallel.
32   *
33   * @param <T>
34   *            type of object to pool
35   */
36  @Slf4j
37  public final class AutoReturnableObjectPool<T>
38      extends GenericObjectPool<AutoReturnable<T>> {
39  
40      private static final ConcurrentHashMap<Class<?>, AutoReturnableObjectPool<?>> POOLS =
41          new ConcurrentHashMap<>();
42  
43      /**
44       * Creates a pool of objects with size of maximum available objects; see
45       * {@link AutoReturnableObjectFactory#AutoReturnableObjectFactory(List)}
46       *
47       * @param factory
48       *            the pooled objects factory
49       */
50      public AutoReturnableObjectPool(
51          final AutoReturnableObjectFactory<T> factory) {
52          super(factory);
53          val total = factory.objects.size();
54          log.debug("setting total pooled objects to {} in pool {}",
55              total, hashCode());
56          setMaxTotal(total);
57          setMaxIdle(-1); // otherwise maxTotal is not respected
58      }
59  
60      /**
61       * Initializes an object pool.
62       *
63       * @param availableObjects
64       *            list of available objects for pooling
65       * @param <T>
66       *            type of objects to be pooled
67       * @return pool for specified type of available objects
68       *
69       * @throws IllegalArgumentException
70       *             if list of available objects is empty
71       */
72      public static <T> AutoReturnableObjectPool<T> poolFor(
73          final List<T> availableObjects) {
74          if (availableObjects.isEmpty())
75              throw new IllegalArgumentException(
76                  "available objects list must contain at least one item");
77          return new AutoReturnableObjectPool<>(
78              new AutoReturnableObjectFactory<>(availableObjects));
79      }
80  
81      /**
82       * Initializes a singleton object pool. Available objects will be fixed at
83       * first call.
84       *
85       * @param availableObjects
86       *            list of available objects for pooling
87       * @param <T>
88       *            type of objects to be pooled
89       * @return singleton pool for specified type of available objects; further
90       *         calls will return same pool with same available object as
91       *         provided at first call
92       *
93       * @throws IllegalArgumentException
94       *             if list of available objects is empty
95       */
96      @SuppressWarnings("unchecked")
97      public static <T> AutoReturnableObjectPool<T> singletonPoolFor(
98          final List<T> availableObjects) {
99          if (availableObjects.isEmpty())
100             throw new IllegalArgumentException(
101                 "available objects list must contain at least one item");
102 
103         return (AutoReturnableObjectPool<T>) POOLS
104             // NOTE: ConcurrentHashMap#computeIfAbsent is atomic
105             // and the creation function below is guaranteed to be called once
106             .computeIfAbsent(
107                 availableObjects.get(0).getClass(),
108                 (clazz) -> poolFor(availableObjects));
109     }
110 
111     /**
112      * Borrows a pooled object and marks it as pooled by this pool.
113      *
114      * @return a pooled object
115      * @throws Exception
116      *             if failed to create new pooled object; see
117      *             {@link AutoReturnableObjectFactory#create()}
118      */
119     @Override
120     public AutoReturnable<T> borrowObject() throws Exception {
121         val pooledObject = super.borrowObject();
122         log.trace("borrowing object {}", pooledObject.$.toString());
123         pooledObject.pool = this;
124         return pooledObject;
125     }
126 }
