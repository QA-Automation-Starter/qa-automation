111     private static Pair<BigInteger, BigInteger> aA(
112         final BigInteger N,
113         final BigInteger g) {
114         BigInteger a, A;
115         do {
116             a = new BigInteger(1024, sha1PRNG()).mod(N);
117             A = g.modPow(a, N);
118         } while (A.mod(N).equals(BigInteger.ZERO));
119         return Pair.of(a, A);
120     }

180     private static byte[] messageFor(final byte[]... byteArrays) {
181         val messageDigest = sha256Digest();
182 
183         for (final byte[] byteArray : byteArrays) {
184             messageDigest.update(byteArray);
185         }
186 
187         return messageDigest.digest();
188     }

200     private static BigInteger u(
201         final BigInteger B,
202         final Pair<BigInteger, BigInteger> aA) {
203         val u = new BigInteger(1,
204             messageFor(aA.getRight().toByteArray(), B.toByteArray()));
205 
206         if (u.equals(BigInteger.ZERO))
207             throw new SecurityException("Hash of A and B cannot be zero");
208 
209         return u;
210     }

237     public RespondToAuthChallengeResult authChallengeResultWithoutCaching(
238         final String userName, final String password) {
239         log.debug(
240             "retrieving authentication challenge result for {} with {}:{}",
241             this, userName, password);
242 
243         val cognitoIdentityProvider = cognitoIdentityProvider();
244 
245         // ISSUE https://github.com/aws/aws-sdk-java/issues/2603
246         // unable to use amazon sdk via corporate proxy with self-signed
247         // certificates
248         val initiateAuthResult = cognitoIdentityProvider
249             .initiateAuth(initiateUserSrpAuthRequest(userName));
250 
251         if (!ChallengeNameType.PASSWORD_VERIFIER.toString()
252             .equals(initiateAuthResult.getChallengeName()))
253             throw new SecurityException("challenge failed");
254 
255         return cognitoIdentityProvider
256             .respondToAuthChallenge(
257                 respondToAuthChallengeRequest(
258                     new AuthenticationChallenge(initiateAuthResult),
259                     password));
260     }

277     @SneakyThrows
278     private byte[] getPasswordAuthenticationKey(
279         final String userId,
280         final String userPassword,
281         final BigInteger B,
282         final BigInteger salt) {
283         // Authenticate the password
284         // u = H(A, B)
285 
286         // x = H(salt | H(poolName | userId | ":" | password))
287 
288         val x = x(userId, userPassword, salt);
289         val S = B.subtract(k(N, g).multiply(g.modPow(x, N)))
290             .modPow(aA.getLeft().add(u(B, aA).multiply(x)), N).mod(N);
291 
292         return new Hkdf(S.toByteArray(), u(B, aA).toByteArray())
293             .deriveKey();
294     }

296     @SneakyThrows
297     private Mac initHmacSha256With(final Key key) {
298         val mac = Mac.getInstance(algorithm.name());
299         mac.init(key);
300         return mac;
301     }

303     private InitiateAuthRequest initiateUserSrpAuthRequest(
304         final String userName) {
305 
306         val initiateAuthRequest = new InitiateAuthRequest();
307         initiateAuthRequest.setAuthFlow(AuthFlowType.USER_SRP_AUTH);
308         initiateAuthRequest.setClientId(clientId);
309 
310         if (isNotBlank(secretKey)) {
311             initiateAuthRequest.addAuthParametersEntry("SECRET_HASH",
312                 secretHashFor(
313                     new SecretKeySpec(utf8BytesFrom(secretKey),
314                         algorithm.name()),
315                     userName));
316         }
317 
318         initiateAuthRequest.addAuthParametersEntry("USERNAME", userName);
319         // noinspection MagicNumber
320         initiateAuthRequest.addAuthParametersEntry("SRP_A",
321             aA.getRight().toString(16));
322         return initiateAuthRequest;
323     }

325     private byte[] macFor(final Key key, final byte[]... byteArrays) {
326         val mac = initHmacSha256With(key);
327 
328         for (final byte[] byteArray : byteArrays) {
329             mac.update(byteArray);
330         }
331 
332         return mac.doFinal();
333     }

335     @SuppressWarnings("serial")
336     @SneakyThrows
337     private RespondToAuthChallengeRequest respondToAuthChallengeRequest(
338         final AuthenticationChallenge challenge,
339         final String password) {
340 
341         val timestamp = challengeTimeFormat().format(new Date());
342 
343         val respondToAuthChallengeRequest = new RespondToAuthChallengeRequest();
344         respondToAuthChallengeRequest.setClientId(clientId);
345         respondToAuthChallengeRequest.setSession(challenge.session());
346         respondToAuthChallengeRequest.setChallengeName(challenge.name());
347         respondToAuthChallengeRequest.setChallengeResponses(
348             new HashMap<String, String>() {
349                 {
350                     put("PASSWORD_CLAIM_SECRET_BLOCK", challenge.secretBlock());
351                     put("USERNAME", challenge.username());
352                     put("TIMESTAMP", timestamp);
353                     put("PASSWORD_CLAIM_SIGNATURE",
354                         utf8StringFrom(
355                             Base64.getEncoder().encode(
356                                 macFor(
357                                     new SecretKeySpec(
358                                         getPasswordAuthenticationKey(
359                                             challenge.userIdForSRP(),
360                                             password,
361                                             challenge.srpB(),
362                                             challenge.salt()),
363                                         algorithm.name()),
364                                     userPoolIdNoise(),
365                                     utf8BytesFrom(challenge.userIdForSRP()),
366                                     Base64.getDecoder()
367                                         .decode(challenge.secretBlock()),
368                                     utf8BytesFrom(timestamp)))));
369                 }
370             });
371 
372         return respondToAuthChallengeRequest;
373     }

375     @SneakyThrows
376     private String secretHashFor(final Key key, final String userName) {
377         return Base64.getEncoder()
378             .encodeToString(
379                 macFor(key, utf8BytesFrom(userName), utf8BytesFrom(clientId)));

382     private byte[] userIdHash(
383         final String userId,
384         final String userPassword) {
385         return messageFor(
386             userPoolIdNoise(),
387             utf8BytesFrom(userId),
388             utf8BytesFrom(COLON),
389             utf8BytesFrom(userPassword));
390     }

392     private byte[] userPoolIdNoise() {
393         return utf8BytesFrom(substringAfter(userPoolId, "_"));
394     }

396     private BigInteger x(
397         final String userId,
398         final String userPassword,
399         final BigInteger salt) {
400         return new BigInteger(1,
401             messageFor(salt.toByteArray(), userIdHash(userId, userPassword)));

418     final static class AuthenticationChallenge {
419         private final InitiateAuthResult challenge;
420 
421         String name() {
422             return challenge.getChallengeName();
423         }
424 
425         BigInteger salt() {
426             // noinspection MagicNumber
427             return new BigInteger(
428                 challenge.getChallengeParameters().get("SALT"),
429                 16);
430         }
431 
432         String secretBlock() {
433             return challenge.getChallengeParameters().get("SECRET_BLOCK");
434         }
435 
436         String session() {
437             return challenge.getSession();
438         }
439 
440         BigInteger srpB() {
441             val srpB = new BigInteger(
442                 challenge.getChallengeParameters().get("SRP_B"),
443                 16);
444 
445             if (srpB.mod(N).equals(BigInteger.ZERO))
446                 throw new SecurityException("SRP error, B cannot be zero");
447 
448             return srpB;
449         }
450 
451         String userIdForSRP() {
452             return challenge.getChallengeParameters().get("USER_ID_FOR_SRP");
453         }
454 
455         String username() {
456             return challenge.getChallengeParameters().get("USERNAME");
457         }

469     @SuppressFBWarnings(value = "IMA_INEFFICIENT_MEMBER_ACCESS",
470         justification = "no real performance impact as this class does I/O")
471     final class Hkdf {
472         static final String DERIVED_KEY_INFO = "Caldera Derived Key";
473         static final int    DERIVED_KEY_SIZE = 16;
474 
475         static final int    MAX_KEY_SIZE     = 255;
476         final byte[]        EMPTY_ARRAY      = new byte[0];
477         final SecretKey     privateKey;
478 
479         @SneakyThrows
480         Hkdf(final byte[] ikm, final byte[] salt) {
481             privateKey = new SecretKeySpec(
482                 initHmacSha256With(new SecretKeySpec(salt, algorithm.name()))
483                     .doFinal(ikm),
484                 algorithm.name());
485         }
486 
487         byte[] deriveKey() {
488             return deriveKey(utf8BytesFrom(DERIVED_KEY_INFO), DERIVED_KEY_SIZE);
489         }
490 
491         @SneakyThrows
492         byte[] deriveKey(final byte[] info, final int length) {
493             val result = new byte[length];
494 
495             val mac = initHmacSha256With(privateKey);
496             if (length > MAX_KEY_SIZE * mac.getMacLength())
497                 throw new IllegalArgumentException(
498                     "Requested keys may not be longer than 255 times the underlying HMAC length.");
499 
500             {
501                 var t = EMPTY_ARRAY;
502                 var loc = 0;
503 
504                 for (byte i = 1; loc < length; i++) {
505                     mac.update(t);
506                     mac.update(info);
507                     mac.update(i);
508                     t = mac.doFinal();
509 
510                     for (int x = 0; x < t.length && loc < length; x++, loc++) {
511                         result[loc] = t[x];
512                     }
513                 }
514             }
515 
516             return result;
517         }
