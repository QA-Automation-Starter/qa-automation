19  import java.lang.reflect.*;
20  import java.util.*;

22  // NOTE: code from https://github.com/lbovet/super-machine/
23  @SuppressWarnings({ "rawtypes", "unchecked" })
24  public class Traverser {
25      private final Map<Object, Object>   _objVisited = new IdentityHashMap<>();
26      private final Map<Class, ClassInfo> _classCache = new HashMap<>();
27  
28      /**
29       * @param o
30       *            Any Java Object
31       * @param skip
32       *            String[] of class names to not include in the tally
33       * @param stopClass
34       *            stopClass
35       * @param visitor
36       *            Visitor is called for every object encountered during the Java
37       *            object graph traverser.
38       */
39      public static void traverse(Object o, Class[] skip, Class stopClass,
40          Visitor visitor) {
41          Traverser traverse = new Traverser();
42          traverse.walk(o, skip, stopClass, visitor);
43          traverse._objVisited.clear();
44          traverse._classCache.clear();
45      }
46  
47      private static void walkCollection(Deque stack, Collection col) {
48          for (Object o : col) {
49              if (o != null && !o.getClass().isPrimitive()) {
50                  stack.add(o);
51              }
52          }
53      }
54

67      private static void walkMap(Deque stack, Map map) {
68          for (Map.Entry entry : (Iterable<Map.Entry>) map.entrySet()) {
69              Object o = entry.getKey();
70  
71              if (o != null && !o.getClass().isPrimitive()) {
72                  stack.add(entry.getKey());
73                  stack.add(entry.getValue());
74              }
75          }
76      }
77  
78      private ClassInfo getClassInfo(Class current, Class[] skip) {
79          ClassInfo classCache = _classCache.get(current);
80          if (classCache != null) {
81              return classCache;
82          }
83  
84          classCache = new ClassInfo(current, skip);
85          _classCache.put(current, classCache);
86          return classCache;
87      }
88  
89      /**
90       * Traverse the object graph referenced by the passed in root.
91       *
92       * @param root
93       *            Any Java object.
94       * @param skip
95       *            Set of classes to skip (ignore). Allowed to be null.
96       */
97      private void walk(Object root, Class[] skip, Class stopClass,
98          Visitor visitor) {
99          Deque stack = new LinkedList();
100         stack.add(root);
101 
102         while (!stack.isEmpty()) {
103             Object current = stack.removeFirst();
104 
105             if (current == null || _objVisited.containsKey(current)) {
106                 continue;
107             }
108 
109             final Class clazz = current.getClass();
110             ClassInfo classInfo = getClassInfo(clazz, skip);
111             if (classInfo._skip) { // Do not process any classes that are
112                                    // assignableFrom the skip classes list.
113                 continue;
114             }
115 
116             _objVisited.put(current, null);
117             visitor.process(current);
118 
119             if (stopClass == null || clazz != stopClass) {
120                 if (clazz.isArray()) {
121                     int len = Array.getLength(current);
122                     Class compType = clazz.getComponentType();
123 
124                     if (!compType.isPrimitive()) { // Speed up: do not walk
125                                                    // primitives
126                         ClassInfo info = getClassInfo(compType, skip);
127                         if (!info._skip) { // Do not walk array elements of a
128                                            // class type that is to be skipped.
129                             for (int i = 0; i < len; i++) {
130                                 Object element = Array.get(current, i);
131                                 if (element != null) { // Skip processing null
132                                                        // array elements
133                                     stack.add(Array.get(current, i));
134                                 }
135                             }
136                         }
137                     }
138                 } else { // Process fields of an object instance
139                     if (current instanceof Collection) {
140                         walkCollection(stack, (Collection) current);
141                     } else if (current instanceof Map) {
142                         walkMap(stack, (Map) current);
143                     } else {
144                         walkFields(stack, current, skip);
145                     }
146                 }
147             }
148         }
149     }
150 
151     private void walkFields(Deque stack, Object current, Class[] skip) {
152         ClassInfo classInfo = getClassInfo(current.getClass(), skip);
153 
154         for (Field field : classInfo._refFields) {
155             try {
156                 Object value = field.get(current);
157                 if (value == null || value.getClass().isPrimitive()) {
158                     continue;
159                 }
160                 stack.add(value);
161             } catch (@SuppressWarnings("unused") IllegalAccessException ignored) {
162             }
163         }
164     }
165 
166     public interface Visitor {
167         void process(Object o);
168     }
169 
170     /**
171      * This class wraps a class in order to cache the fields so they are only
172      * reflectively obtained once.
173      */
174     private static class ClassInfo {
175         private final Collection<Field> _refFields = new ArrayList<>();
176         private boolean                 _skip      = false;
177 
178         public ClassInfo(Class c, Class[] skip) {
179             if (skip != null) {
180                 for (Class klass : skip) {
181                     if (klass.isAssignableFrom(c)) {
182                         _skip = true;
183                         return;
184                     }
185                 }
186             }
187 
188             Collection<Field> fields = ReflectionUtils.getDeepDeclaredFields(c);
189             for (Field field : fields) {
190                 Class fc = field.getType();
191 
192                 if (!fc.isPrimitive()) {
193                     _refFields.add(field);
194                 }
195             }
196         }
197     }
