17  package dev.aherscu.qa.testing.extra.supermachine.traverser;
18  
19  import java.lang.SuppressWarnings;
20  import java.lang.reflect.*;
21  import java.util.*;
22  import java.util.concurrent.*;
23  
24  import edu.umd.cs.findbugs.annotations.*;
25  
26  // NOTE: code from https://github.com/lbovet/super-machine/
27  @SuppressWarnings({ "rawtypes", "unchecked", "NestedTryStatement" })
28  public final class ReflectionUtils {
29      private static final Map<Class, Collection<Field>> _reflectedFields =
30          new ConcurrentHashMap<>();
31  
32      private ReflectionUtils() {
33          super();
34      }
35  
36      /**
37       * Get all non static, non transient, fields of the passed in class,
38       * including private fields. Note, the special this$ field is also not
39       * returned. The resulting fields are stored in a Collection.
40       *
41       * @param c
42       *            Class instance that would need further processing (reference
43       *            fields). This makes field traverser on a class faster as it
44       *            does not need to continually process known fields like
45       *            primitives.
46       * @param fields
47       *            fields
48       */
49      @SuppressFBWarnings(
50          value = { "DE_MIGHT_IGNORE", "DP_DO_INSIDE_DO_PRIVILEGED" },
51          justification = "library code")
52      public static void getDeclaredFields(Class c, Collection<Field> fields) {
53          try {
54              Field[] local = c.getDeclaredFields();
55  
56              for (Field field : local) {
57                  if (!field.isAccessible()) {
58                      try {
59                          field.setAccessible(true);
60                      } catch (@SuppressWarnings("unused") Exception ignored) {
61                      }
62                  }
63  
64                  int modifiers = field.getModifiers();
65                  if (!Modifier.isStatic(modifiers) &&
66                      !field.getName().startsWith("this$") &&
67                      !Modifier.isTransient(modifiers)) { // speed up: do not
68                                                          // count static fields,
69                                                          // do not go back up to
70                                                          // enclosing object in
71                                                          // nested case, do not
72                                                          // consider transients
73                      fields.add(field);
74                  }
75              }
76          } catch (@SuppressWarnings("unused") Throwable ignored) {
77              // ignored
78          }
79      }
80  
81      /**
82       * Get all non static, non transient, fields of the passed in class,
83       * including private fields. Note, the special this$ field is also not
84       * returned. The result is cached in a static ConcurrentHashMap to benefit
85       * execution performance.
86       *
87       * @param c
88       *            Class instance
89       * @return Collection of only the fields in the passed in class that would
90       *         need further processing (reference fields). This makes field
91       *         traverser on a class faster as it does not need to continually
92       *         process known fields like primitives.
93       */
94      public static Collection<Field> getDeepDeclaredFields(Class c) {
95          if (_reflectedFields.containsKey(c)) {
96              return _reflectedFields.get(c);
97          }
98          Collection<Field> fields = new ArrayList<>();
99          Class curr = c;
100 
101         while (curr != null) {
102             getDeclaredFields(curr, fields);
103             curr = curr.getSuperclass();
104         }
105         _reflectedFields.put(c, fields);
106         return fields;
107     }
108 
109     @SuppressFBWarnings("REC_CATCH_EXCEPTION")
110     public static Method getMethod(Class c, String method, Class... types) {
111         try {
112             return c.getMethod(method, types);
113         } catch (@SuppressWarnings("unused") Exception nse) {
114             return null;
115         }
116     }
117 }
