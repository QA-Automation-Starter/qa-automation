17  package dev.aherscu.qa.testing.extra.supermachine;
18  
19  import java.util.*;
20  import java.util.stream.*;
21  
22  import dev.aherscu.qa.testing.extra.supermachine.traverser.*;
23  
24  /**
25   * A scanner traversing an object graph.
26   *
27   * @param <T>
28   *            the type of the traversed items.
29   */
30  public final class BeanScanner<T> extends Scanner<T> {
31  
32      private BeanScanner(Stream<T> stream) {
33          super(stream);
34      }
35  
36      /**
37       * Creates a bean scanner.
38       *
39       * @param root
40       *            the root object from which the graph will be traversed.
41       * @param <X>
42       *            the type of the root object.
43       * @return a scanner for the root object.
44       */
45      public static <X> Scanner<X> from(X root) {
46          return new BeanScanner<>(Stream.of(root));
47      }
48  
49      /**
50       * Traverses the object graph for each items of this scanner and returns a
51       * scanner emitting the first item of the given type found in each branch.
52       *
53       * @param clazz
54       *            the type selector.
55       * @param <X>
56       *            type of the selector and resulting items.
57       * @return a scanner emitting the selected items.
58       */
59      @Override
60      public <X> Scanner<X> find(Class<X> clazz) {
61          return new BeanScanner<>(stream().flatMap(x -> traverse(x, clazz)))
62              .superFind(clazz);
63      }
64  
65      /**
66       * Traverses the object graph for each items of this scanner and returns a
67       * scanner emitting all found items of the given type.
68       *
69       * @param clazz
70       *            the type selector.
71       * @param <X>
72       *            type of the selector and resulting items.
73       * @return a scanner emitting the selected items.
74       */
75      @Override
76      public <X> Scanner<X> walk(Class<X> clazz) {
77          return new BeanScanner<>(stream().flatMap(x -> traverse(x, null)))
78              .superFind(clazz);
79      }
80  
81      @Override
82      protected <X> Scanner<X> create(Stream<X> source) {
83          return new BeanScanner<>(source);
84      }
85  
86      private <X> Scanner<X> superFind(Class<X> clazz) {
87          return super.walk(clazz);
88      }
89  
90      private <X> Stream<Object> traverse(T root, Class<X> stopClass) {
91          LinkedList<Object> list = new LinkedList<>();
92          Traverser.traverse(root, null, stopClass, list::add);
93          return list.stream();
94      }
95  }
