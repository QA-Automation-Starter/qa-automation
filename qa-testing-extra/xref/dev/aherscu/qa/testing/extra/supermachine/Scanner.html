17  package dev.aherscu.qa.testing.extra.supermachine;
18  
19  import java.util.function.*;
20  import java.util.stream.*;
21  
22  /**
23   * Abstraction wrapping a data structure that can be walked through to produce
24   * streams.
25   * 
26   * @param <T>
27   *            the type of elements composing the data structure.
28   */
29  public abstract class Scanner<T> {
30  
31      private final Stream<T> stream;
32  
33      /**
34       * Constructs a scanner out of a source stream.
35       *
36       * @param source
37       *            the wrapped stream, will contain only one element when it is a
38       *            root.
39       */
40      protected Scanner(Stream<T> source) {
41          this.stream = source;
42      }
43  
44      /**
45       * Apply a function to every items of this scanner and returns a scanner
46       * emitting the non-null results.
47       *
48       * @param extractor
49       *            the function applied to each item, can return null.
50       * @param <X>
51       *            the returned type.
52       * @return a scanner emitting non-null results of the function applied to
53       *         items.
54       */
55      public <X> Scanner<X> extract(Function<T, X> extractor) {
56          return create(stream.flatMap(x -> {
57              X result = extractor.apply(x);
58              return result != null ? Stream.of(result) : Stream.empty();
59          }));
60      }
61  
62      /**
63       * Filters out items according to predicate.
64       *
65       * @param predicate
66       *            the condition to emit the items.
67       * @return a scanner emitting the items matching the predicate.
68       */
69      public Scanner<T> filter(Predicate<T> predicate) {
70          return create(stream.filter(predicate));
71      }
72  
73      /**
74       * Scans the elements of the source and returns a scanner emitting relevant
75       * items of the given type. This method can return fewer items according to
76       * subclass logic. Subclasses implement the actual scanning algorithm.
77       * 
78       * @param clazz
79       *            the type selector.
80       * @param <X>
81       *            type of the selector and resulting items.
82       * @return a scanner emitting the selected items.
83       */
84      public <X> Scanner<X> find(Class<X> clazz) {
85          return walk(clazz);
86      }
87  
88      /**
89       * Extracts the items of this scanner as a stream.
90       *
91       * @return a stream of items.
92       */
93      public Stream<T> stream() {
94          return stream;
95      }
96  
97      /**
98       * Transforms this scanner into one or more derived scanners and merges
99       * their emitted items.
100      *
101      * @param fns
102      *            the functions that will be applied to this scanner and produce
103      *            the scanners to merge.
104      * @param <X>
105      *            the common type of items resulting of the scanners to merge.
106      * @return a scanner merging the result of the composed scanners.
107      */
108     @SafeVarargs
109     final public <X> Scanner<X> then(Function<Scanner<T>, Scanner<X>>... fns) {
110         return create(stream.flatMap(x -> Stream.of(fns)
111             .map(fn -> fn.apply(create(Stream.of(x))).stream())
112             .reduce(Stream.empty(), Stream::concat)));
113     }
114 
115     /**
116      * Scans the elements of the source and returns a scanner emitting all
117      * resulting items of the given type. Subclasses implement the actual
118      * scanning algorithm.
119      *
120      * @param clazz
121      *            the type selector.
122      * @param <X>
123      *            type of the selector and resulting items.
124      * @return a scanner emitting the selected items.
125      */
126     public <X> Scanner<X> walk(Class<X> clazz) {
127         @SuppressWarnings("unchecked")
128         Stream<X> filteredStream = (Stream<X>) stream
129             .filter(x -> clazz.isAssignableFrom(x.getClass()));
130         return create(filteredStream);
131     }
132 
133     /**
134      * Constructs a scanner out of a source stream.
135      *
136      * @param source
137      *            the wrapped stream, will contain only one element when it is a
138      *            root.
139      * @param <X>
140      *            the type of elements in the source stream.
141      * @return a scanner based on the given source.
142      */
143     protected abstract <X> Scanner<X> create(Stream<X> source);
144 }
