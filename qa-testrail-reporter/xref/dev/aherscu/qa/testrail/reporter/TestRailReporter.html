43  /**
44   * Per method test reporter uploading results with screenshot attachments to
45   * TestRail.
46   *
47   * @see #with(XmlSuite) support for TestNG Listener configuration
48   */
49  @SuperBuilder(toBuilder = true)
50  @NoArgsConstructor(force = true)
51  @Slf4j
52  @ToString(callSuper = true)
53  public class TestRailReporter extends QaJGivenPerMethodReporter {
54      // FIXME use .mustache extension
55      @SuppressWarnings("hiding")
56      public static final String DEFAULT_TEMPLATE_RESOURCE =
57          "/permethod-reporter.testrail";
58  
59      private final URI          testRailUrl;
60      private final String       testRailRunId;
61  
62      private static TestRailClient testRailClient(final URI testRailUrl) {
63          val testRailClient = new TestRailClient(testRailUrl.toString());
64          testRailClient.setUser(usernameFrom(testRailUrl));
65          testRailClient.setPassword(passwordFrom(testRailUrl));
66          return testRailClient;
67      }
68  
69      private static Collection<File> listScreenshots(final File directory) {
70          return listFiles(directory, new SuffixFileFilter(".png"), null);
71      }
72  
73      /**
74       * Builds a new reporter configured with additional TestNG XML suite
75       * parameters:
76       * <dl>
77       * <dt>testRailRunId</dt>
78       * <dd>the TestRail Run to report to</dd>
79       * <dt>testRailUrl</dt>
80       * <dd>the TestRail location</dd>
81       * </dl>
82       * 
83       * @param xmlSuite
84       *            TestNG XML suite
85       * @return reporter configured
86       */
87      @Override
88      protected TestRailReporter with(final XmlSuite xmlSuite) {
89          return ((TestRailReporter) super.with(xmlSuite))
90              .toBuilder()
91              .templateResource(templateResourceParamFrom(xmlSuite,
92                  DEFAULT_TEMPLATE_RESOURCE))
93              .testRailRunId(
94                  requireNonNull(xmlSuite.getParameter("testRailRunId"),
95                      "testRailRunId parameter not found in current TestNG XML"))
96              .testRailUrl(URI.create(
97                  requireNonNull(xmlSuite.getParameter("testRailUrl"),
98                      "testRailUrl parameter not found in current TestNG XML")))
99              .build();
100     }
101 
102     @Override
103     protected Mustache.Compiler compiler() {
104         return Mustache.compiler().withEscaper(Escapers.NONE);
105     }
106 
107     private AttachScreenshotsResponse addAttachmentToResult(
108         final String resultId,
109         final File screenshot) {
110         return fromJson(testRailClient(testRailUrl)
111             .sendPost(format("add_attachment_to_result/{0}",
112                 resultId),
113                 screenshot.toString())
114             .toString(),
115             AttachScreenshotsResponse.class);
116     }
117 
118     private ResultForCaseResponse addResultForCase(
119         final ScenarioModel scenarioModel,
120         final File reportFile, final String testCaseId) throws IOException {
121         try (val fileReader = new FileReader(reportFile)) {
122             return fromJson(testRailClient(testRailUrl)
123                 .sendPost(format("add_result_for_case/{0}/{1}",
124                     testRailRunId,
125                     testCaseId),
126                     ImmutableMap.builder()
127                         .put("status_id",
128                             Status.from(scenarioModel.getExecutionStatus()).id)
129                         .put("comment",
130                             IOUtils.toString(fileReader))
131                         .build())
132                 .toString(),
133                 ResultForCaseResponse.class);
134         }
135     }
136 
137     @Override
138     protected void reportGenerated(
139         final ScenarioModel scenarioModel,
140         final File reportFile) {
141         super.reportGenerated(scenarioModel, reportFile);
142 
143         val testCaseId = readAttributesOf(reportFile)
144             // TODO make it configurable
145             .get("dev.aherscu.qa.jgiven.commons.tags.Reference");
146 
147         try {
148             val addResultForCaseResponse =
149                 addResultForCase(scenarioModel, reportFile, testCaseId);
150             log.debug(
151                 "reported result id {} for case {} on run {} to {}/index.php?/tests/view/{}",
152                 addResultForCaseResponse.id,
153                 testCaseId, testRailRunId, testRailUrl,
154                 addResultForCaseResponse.testId);
155 
156             // FIXME for each report only its screenshots must be attached
157             // TODO the screenshots must be somehow associated with their report
158             listScreenshots(
159                 new File(outputDirectory, targetNameFor(scenarioModel)))
160                     .forEach(file -> {
161                         log.trace("attaching {}", file);
162                         val attachScreenshotsResponse =
163                             addAttachmentToResult(addResultForCaseResponse.id,
164                                 file);
165                         log.debug("attached {}", attachScreenshotsResponse.id);
166                     });
167 
168         } catch (final Exception e) {
169             log.error("failed to report case {} on run {} -> {}",
170                 testCaseId, testRailRunId, e.getMessage());
171         }
172     }
173 
174     @Override
175     protected TestRailReportModel reportModel(File targetReportFile) {
176         return TestRailReportModel.builder()
177             .outputDirectory(outputDirectory)
178             .targetReportFile(targetReportFile)
179             .build();
180     }
181 
182     @Getter
183     @AllArgsConstructor
184     private enum Status {
185         SUCCESS(1), FAILED(5);
186 
187         final int id;
188 
189         static Status from(final ExecutionStatus status) {
190             return status == ExecutionStatus.SUCCESS ? SUCCESS : FAILED;
191         }
192     }
193 
194     @JsonIgnoreProperties(ignoreUnknown = true)
195     static class ResultForCaseResponse {
196         @JsonProperty("id")
197         String id;
198         @JsonProperty("test_id")
199         String testId;
200     }
201 
202     @JsonIgnoreProperties(ignoreUnknown = true)
203     static class AttachScreenshotsResponse {
204         @JsonProperty("attachment_id")
205         String id;
206     }
