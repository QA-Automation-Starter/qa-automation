37  /**
38   * Contains REST sample tests just to ensure that the testing infrastructure
39   * works as required.
40   *
41   * @author aherscu
42   *
43   */
44  @SelfTest
45  @RestTest
46  @edu.umd.cs.findbugs.annotations.SuppressFBWarnings(
47      value = "BC_UNCONFIRMED_CAST_OF_RETURN_VALUE",
48      justification = "JGiven framework limitation")
49  @SuppressWarnings({ "boxing" })
50  public final class StatefulRestTest
51      extends AbstractWireMockTest<RestScenarioType, RestFixtures<?>, RestActions<?>, RestVerifications<?>> {
52  
53      private Client client;
54  
55      /**
56       * Should retrieve a JSON field from a REST service
57       */
58      @Test
59      public void shouldAuthenticate() {
60          given()
61              .a_REST_client(client);
62  
63          when()
64              .connecting_to(wireMockServer.baseUrl())
65              .and().appending_path("authenticate")
66              .and().posting(new Form());
67  
68          then()
69              .the_response_status(is(Response.Status.Family.SUCCESSFUL));
70      }
71  
72      @Test(dependsOnMethods = "shouldOperate")
73      public void shouldFailOnAuthentication() {
74          given()
75              .a_REST_client(client);
76  
77          when()
78              .connecting_to(wireMockServer.baseUrl())
79              .and().appending_path("some-id")
80              .and().getting_the_response();
81  
82          then()
83              .the_response_status(is(Response.Status.Family.CLIENT_ERROR));
84      }
85  
86      @Test(dependsOnMethods = "shouldAuthenticate")
87      public void shouldOperate() {
88          given()
89              .a_REST_client(client);
90  
91          when()
92              .connecting_to(wireMockServer.baseUrl())
93              .and().appending_path("some-id")
94              .and().getting_the_response();
95  
96          then()
97              .the_response_status(is(Response.Status.Family.SUCCESSFUL))
98              .and().the_response_contents(asList(
99                  new JsonAssertion<>("$[0].id", greaterThan(0)),
100                 new JsonAssertion<>("$[1].id", equalTo(2))));
101     }
102 
103     @AfterClass
104     private void afterClassCloseRestClient() {
105         client.close();
106     }
107 
108     @BeforeClass
109     private void beforeClassAddStubs() {
110         val AUTHENTICATED = "AUTHENTICATED";
111         wireMockServer.stubFor(post(urlEqualTo("/authenticate"))
112             .inScenario(StatefulRestTest.class.getName())
113             .willReturn(ok())
114             .willSetStateTo(AUTHENTICATED));
115 
116         wireMockServer.stubFor(get(urlEqualTo("/some-id"))
117             .inScenario(StatefulRestTest.class.getName())
118             .whenScenarioStateIs(AUTHENTICATED)
119             .willReturn(ok("[{id:1},{id:2},{id:3}]"))
120             .willSetStateTo(STARTED));
121 
122         wireMockServer.stubFor(get(urlEqualTo("/some-id"))
123             .inScenario(StatefulRestTest.class.getName())
124             .whenScenarioStateIs(STARTED)
125             .willReturn(unauthorized())
126             .willSetStateTo(STARTED));
127     }
128 
129     @BeforeClass
130     private void beforeClassOpenRestClient() {
131         client = LoggingClientBuilder.newClient();
132     }
133 }
