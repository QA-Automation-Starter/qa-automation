17  package dev.aherscu.qa.testing.example.steps.tutorial;
18  
19  import static jakarta.ws.rs.client.Entity.*;
20  
21  import com.tngtech.jgiven.annotation.*;
22  
23  import dev.aherscu.qa.jgiven.commons.model.*;
24  import dev.aherscu.qa.jgiven.rest.steps.*;
25  import dev.aherscu.qa.testing.example.*;
26  import dev.aherscu.qa.testing.example.model.tutorial.*;
27  
28  public class BinanceActions<SELF extends BinanceActions<SELF>>
29      extends RestActions<SELF> {
30  
31      @ExpectedScenarioState
32      protected TestConfiguration                       configuration;
33  
34      @ProvidedScenarioState
35      protected final ThreadLocal<SwaggerLoginResponse> loginResponse =
36          new ThreadLocal<>();
37  
38      public SELF adding(final Pet pet) {
39          return retrying(invoke(configuration.petStore(client.get())
40              .path("pet")
41              .request()
42              .buildPost(json(pet))));
43      }
44  
45      public SELF logging_in_with(final Credentials credentials) {
46          return retrying(invoke(configuration.petStore(client.get())
47              .path("user/login")
48              .queryParam("username", credentials.userName)
49              .queryParam("password", credentials.password)
50              .request()
51              .buildGet(),
52              json -> loginResponse.set(SwaggerLoginResponse.from(json))));
53      }
54  }
