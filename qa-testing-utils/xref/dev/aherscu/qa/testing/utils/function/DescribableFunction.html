3   import static dev.aherscu.qa.testing.utils.internal.DescriptionUtils.withPrefixedArticle;
4   
5   import java.lang.invoke.MethodType;
6   import java.lang.invoke.SerializedLambda;
7   import java.util.function.Function;
8   
9   @FunctionalInterface
10  public interface DescribableFunction<T, R> extends Function<T, R>,
11      SerializedLambdaResolvable, SingleArgumentDescribableFunctionalInterface {
12  
13      default String getResultDescription() {
14          SerializedLambda lambda = asSerializedLambda();
15          MethodType lambdaMethodType = MethodType.fromMethodDescriptorString(
16              lambda.getImplMethodSignature(), getClass().getClassLoader());
17          String resultType = lambdaMethodType.returnType().getSimpleName();
18          if (!lambda.getImplMethodName().startsWith("lambda$")) {
19              return lambda.getImplMethodName() + " ("
20                  + withPrefixedArticle(resultType) + ")";
21          }
22          return resultType;
23      }
24  
25  }
