30  /**
31   * Represents a configuration.
32   *
33   * @author aherscu
34   *
35   */
36  @Slf4j
37  public class BaseConfiguration
38      extends AbstractConfiguration<CompositeConfiguration> {
39  
40      /**
41       * Loads the specified configurations.
42       *
43       * @param configurations
44       *            the additional configurations; might be null or empty
45       */
46      public BaseConfiguration(final Configuration... configurations) {
47          super(new CompositeConfiguration(Arrays.asList(configurations)));
48          log.trace("configuration loaded {}", toString(CR + LF));
49      }
50  
51      /**
52       * Given following properties:
53       *
54       * <pre>
55       * capability1.class           =ANDROID
56       * capability1.device.name     =Samsung Galaxy S9 Plus WQHD
57       * capability1.platform.version=8.0
58       * capability1.appium.version  =1.13.0
59       *
60       * capability2.class           =ANDROID
61       * capability2.device.name     =Samsung Galaxy S9 Plus HD
62       * capability2.platform.version=8.1
63       * capability2.appium.version  =1.13.0
64       * </pre>
65       *
66       * calling {@code groupsOf("capability")} will yield two groups as
67       * following:
68       *
69       * <pre>
70       * class           =ANDROID
71       * device.name     =Samsung Galaxy S9 Plus WQHD
72       * platform.version=8.0
73       * appium.version  =1.13.0
74       * </pre>
75       *
76       * <pre>
77       * class           =ANDROID
78       * device.name     =Samsung Galaxy S9 Plus HD
79       * platform.version=8.1
80       * appium.version  =1.13.0
81       * </pre>
82       *
83       * @param prefix
84       *            the prefix of a group of properties
85       * @return groups of properties, where each group is identified by the
86       *         characters following the prefix up until first dot
87       */
88      public Stream<Map<String, String>> groupsOf(final String prefix) {
89          return entrySet()
90              .stream()
91              .map(entry -> immutableEntry(
92                  entry.getKey().toString(),
93                  entry.getValue().toString()))
94              .filter(entry -> entry.getKey().startsWith(prefix))
95              .collect(groupingBy(
96                  entry -> substringBetween(entry.getKey(), prefix, DOT)))
97              .values()
98              .stream()
99              .map(groupedEntries -> groupedEntries.stream()
100                 .collect(toMap(
101                     groupedEntry -> substringAfter(
102                         substringAfter(groupedEntry.getKey(), prefix), DOT),
103                     Entry::getValue)));
104     }
105 
106     /**
107      * @param delimiter
108      *            character(s) to delimit between entries
109      * @return string representation of all entries, where each entry is
110      *         abbreviated to 127 characters.
111      */
112     public String toString(final CharSequence delimiter) {
113         return entrySet()
114             .stream()
115             .map(Object::toString)
116             .map(s -> abbreviate(s, 127))
117             .sorted()
118             .collect(joining(delimiter));
119     }
120 
121     @Override
122     public String toString() {
123         return toString(SEMI);
124     }
125 }
