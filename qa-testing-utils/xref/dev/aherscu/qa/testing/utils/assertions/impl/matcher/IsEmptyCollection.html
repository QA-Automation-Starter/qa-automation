30  package dev.aherscu.qa.testing.utils.assertions.impl.matcher;
31  
32  import java.util.Collection;
33  
34  import org.hamcrest.Description;
35  import org.hamcrest.Matcher;
36  
37  /**
38   * Tests if collection is empty.
39   */
40  public class IsEmptyCollection<E> extends CollectionMatcher<Collection<E>> {
41  
42      @Override
43      public boolean matchesSafely(Collection<E> item) {
44          return item.isEmpty();
45      }
46  
47      @Override
48      public void describeTo(Description description) {
49          description.appendText("an empty collection");
50      }
51  
52      /**
53       * Matches an empty collection.
54       */
55      public static <E> Matcher<Collection<E>> empty() {
56          return new IsEmptyCollection<E>();
57      }
58  }
