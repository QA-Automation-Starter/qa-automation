18  import static dev.aherscu.qa.testing.utils.StringUtilsExtensions.*;
19  
20  import java.io.*;
21  
22  import com.google.common.base.*;
23  
24  import lombok.*;
25  import lombok.extern.slf4j.*;
26  import net.schmizz.sshj.*;
27  import net.schmizz.sshj.common.*;
28  
29  /**
30   * Encapsulates the {@code ssh-info} field as defined in
31   * <a href="https://tools.ietf.org/html/draft-ietf-secsh-scp-sftp-ssh-uri-04">
32   * Uniform Resource Identifier (URI) Scheme for Secure File Transfer Protocol
33   * (SFTP) and Secure Shell (SSH)</a>.
34   *
35   * @author aherscu
36   */
37  @Slf4j
38  @AllArgsConstructor
39  public final class SshInfo {
40      /**
41       * The default SSH directory; that's the {@code .ssh} directory under the
42       * user's home directory.
43       */
44      public static final File   DEFAULT_SSH_DIR      =
45          new File(StandardSystemProperty.USER_HOME.value(), ".ssh");           //$NON-NLS-1$
46      /**
47       * If you wish to specify a custom SSH private key location then you should
48       * launch your process with
49       * {@code -Dssh.private.key.file=<your private key path>}. This system
50       * property will be consulted before trying to load an SSH private key from
51       * its default location (as specified in {@link #DEFAULT_SSH_DIR}).
52       *
53       * @see #authenticate(SSHClient)
54       */
55      public static final String SSH_PRIVATE_KEY_FILE = "ssh.private.key.file"; //$NON-NLS-1$
56      private final String       sshInfo;
57  
58      private static String defaultSshPrivateKeyFile(final String name) {
59          return new File(DEFAULT_SSH_DIR, name).toString();
60      }
61  
62      private static String sshPrivateKeyFile(final String defaultName) {
63          val file = System.getProperty(SSH_PRIVATE_KEY_FILE,
64              defaultSshPrivateKeyFile(defaultName));
65          log.debug("using SSH private key from {}", file); //$NON-NLS-1$
66          return file;
67      }
68  
69      /**
70       * Authenticates an SSH client using this ssh-info.
71       *
72       * <p>
73       * The recognized {@code ssh-info} structure is as follows: {@code
74       * <user-name>;auth=<method>:<password>}
75       * </p>
76       *
77       * <p>
78       * where:
79       * </p>
80       * <ul>
81       * <li>{@code <user-name>} -- required, the user name</li>
82       * <li>auth={@code <method>} -- required, the authentication method; one of:
83       * <ul>
84       * <li>PASSWORD
85       * <li>PUBLIC_KEY; if this method is used, then an attempt to read an user
86       * defined SSH private key file will be made using the
87       * {@value #SSH_PRIVATE_KEY_FILE} system property, if that fails then it
88       * will fallback on reading {@code id_rsa} file from the
89       * {@link #DEFAULT_SSH_DIR}
90       * </ul>
91       * </li>
92       * <li>{@code <password>} -- required, the password; can be empty
93       * </ul>
94       *
95       *
96       * @param client
97       *            the SSH client
98       * @return the authenticated SSH client
99       * @throws SSHException
100      *             authentication or transport failure
101      */
102     public SSHClient authenticate(final SSHClient client)
103         throws SSHException {
104         switch (authenticationMethod()) {
105         case PASSWORD:
106             client.authPassword(userName(), password());
107             break;
108         case PUBLIC_KEY:
109             client.authPublickey(userName(), sshPrivateKeyFile("id_rsa")); //$NON-NLS-1$
110             break;
111         default:
112             throw new UnsupportedOperationException(sshInfo);
113         }
114         return client;
115     }
116 
117     private SshInfo.AuthMethod authenticationMethod() {
118         return AuthMethod.valueOf(cparam("auth")); //$NON-NLS-1$
119     }
120 
121     private String cparam(final String cparamName) {
122         return Splitter
123             .on(COMMA)
124             .trimResults()
125             .withKeyValueSeparator(EQUAL)
126             .split(substringBetween(sshInfo, SEMI, COLON))
127             .get(cparamName);
128     }
129 
130     private String password() {
131         return substringAfter(sshInfo, COLON);
132     }
133 
134     private String userName() {
135         return substringBefore(substringBefore(sshInfo, COLON), SEMI);
136     }
137 
138     private enum AuthMethod {
139         PASSWORD, PUBLIC_KEY
140     }
141 }
