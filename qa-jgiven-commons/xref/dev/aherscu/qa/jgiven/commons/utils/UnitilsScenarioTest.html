180     @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
181     private static void throwException(final Throwable throwable) {
182         if (throwable instanceof RuntimeException)
183             throw (RuntimeException) throwable;
184         else if (throwable instanceof Error)
185             throw (Error) throwable;
186         else
187             throw new RuntimeException(throwable);
188     }

190     /**
191      * Implementation of the hookable interface to be able to call
192      * {@link TestListener#beforeTestMethod} and
193      * {@link TestListener#afterTestMethod}.
194      *
195      * @param callBack
196      *            the TestNG test callback, not null
197      * @param testResult
198      *            the TestNG test result, not null
199      */
200     // implementation copied from UnitilsTestNG
201     @Override
202     public final void run(
203         final IHookCallBack callBack,
204         final ITestResult testResult) {
205         // ISSUE seems that UnitilsScenarioTest derived test classes
206         // prevents IHookable#run to be invoked, hence cannot use this method
207         // for handling pre/post method execution.
208         // UnitilsScenarioTest implements IHookable#run in order to integrate
209         // with the Unitils framework; perhaps this implementation is faulty.
210 
211         log.trace("running {}", testResult.getName());
212 
213         Throwable beforeTestMethodException = null;
214         try {
215             getTestListener()
216                 .beforeTestMethod(this,
217                     testResult.getMethod().getConstructorOrMethod()
218                         .getMethod());
219 
220         } catch (final Throwable e) {
221             // hold exception until later, first call afterTestMethod
222             beforeTestMethodException = e;
223         }
224 
225         Throwable testMethodException = null;
226         if (beforeTestMethodException == null) {
227             callBack.runTestMethod(testResult);
228 
229             // Since TestNG calls the method using reflection, the exception is
230             // wrapped in an InvocationTargetException
231             testMethodException = testResult.getThrowable();
232             if (testMethodException instanceof InvocationTargetException) {
233                 testMethodException =
234                     ((InvocationTargetException) testMethodException)
235                         .getTargetException();
236             }
237         }
238 
239         Throwable afterTestMethodException = null;
240         try {
241             getTestListener()
242                 .afterTestMethod(
243                     this,
244                     testResult.getMethod().getConstructorOrMethod().getMethod(),
245                     beforeTestMethodException != null
246                         ? beforeTestMethodException
247                         : testMethodException);
248 
249         } catch (final Throwable e) {
250             afterTestMethodException = e;
251         }
252 
253         // if there were exceptions, make sure the exception that occurred first
254         // is reported by TestNG
255         if (beforeTestMethodException != null) {
256             throwException(beforeTestMethodException);
257         } else {
258             // We don't throw the testMethodException, it is already registered
259             // by TestNG and will be reported to the user
260             if (testMethodException == null
261                 && afterTestMethodException != null) {
262                 throwException(afterTestMethodException);
263             }
264         }
265     }

267     /**
268      * On first call initializes the configuration for this scenario type and
269      * caches it; on further calls will return the cached configuration.
270      *
271      * @return the configuration for this type
272      *
273      * @throws RuntimeException
274      *             if the configuration type cannot be instantiated or the
275      *             {@value AbstractConfiguration#CONFIGURATION_SOURCES} refer to
276      *             a non-existing resource.
277      */
278     @SuppressWarnings("unchecked")
279     protected C configuration() {
280         return (C) CACHED_CONFIGURATIONS.computeIfAbsent(
281             configurationType,
282             Unchecked.function((type) -> type
283                 .getConstructor(Configuration[].class)
284                 .newInstance((Object) new Configuration[] {
285                     defaultConfiguration() })));
286     }

309     /**
310      * Called after all test tear down. This is where
311      * {@link TestListener#afterTestTearDown} is called.
312      * <p/>
313      * NOTE: alwaysRun is enabled to be sure that this method is called even
314      * when an exception occurs during {@link #unitilsBeforeMethod}.
315      *
316      * @param testMethod
317      *            The test method, not null
318      */
319     @AfterMethod(alwaysRun = true)
320     protected final void unitilsAfterMethod(final Method testMethod) {
321         // alwaysRun is enabled, extra test to ensure that
322         // unitilsBeforeTestSetUp was called
323         try {
324             if (beforeTestSetUpCalled.get().booleanValue()) {
325                 beforeTestSetUpCalled.set(Boolean.FALSE);
326                 // NOTE: sometimes no transaction is started by Unitils, but it
327                 // always tries to commit the transaction associated with
328                 // current test. When trying to commit a not existent
329                 // transaction a NullPointerException occurs. This seems like a
330                 // bug in Unitils; see DefaultUnitilsTransactionManager#137.
331                 getTestListener().afterTestTearDown(this, testMethod);
332             }
333         } catch (final Exception e) {
334             log.warn("after test got {}", e.getMessage()); //$NON-NLS-1$
335         } finally {
336             // IMPORTANT: when running repeated tests in Jenkins we must write
337             // something into the console otherwise connection is lost
338             log.info("test {}:{} ended", //$NON-NLS-1$
339                 testMethod.getDeclaringClass().getName(),
340                 testMethod.getName());
341         }
342     }

344     /**
345      * <p>
346      * Renames current thread as {@code <class-simple-name>}, as returned by
347      * {@link Class#getSimpleName()}. This means that code running in
348      * {@link BeforeClass} or {@link BeforeTest} annotated methods will have its
349      * thread named such this.
350      * </p>
351      * <p>
352      * Finally, notifies {@code Unitils} about being before class via
353      * {@link TestListener#beforeTestClass(Class)} and
354      * {@link TestListener#afterCreateTestObject(Object)}.
355      * </p>
356      */
357     @BeforeClass(alwaysRun = true)
358     protected final void unitilsBeforeClass() {
359         currentThread().setName(SessionName.builder()
360             .className(getClass().getSimpleName())
361             .id(String.valueOf(currentThread().getId()))
362             .build()
363             .toString());
364 
365         // IMPORTANT: when running repeated tests in Jenkins we must write
366         // something into the console otherwise connection is lost
367         log.info("starting test class {}", getClass().getSimpleName());
368 
369         getTestListener().beforeTestClass(this.getClass());
370         getTestListener().afterCreateTestObject(this);
371     }

373     /**
374      * Called before every test method. This is where
375      * {@link TestListener#beforeTestSetUp} is called.
376      *
377      * <p>
378      * If the {@link #randomId} for {@code current thread} was not already set,
379      * then generates one and sets it. The generated random id is URL-friendly.
380      * Thread starting time is stored in {@link #startTime}.
381      * </p>
382      *
383      * <p>
384      * In addition, renames current thread as
385      * {@code <class-simple-name>:<method-name>:<hash-of-params>@<thread-id>},
386      * where:
387      * <dl>
388      * <dt>class-simple-name</dt>
389      * <dd>as returned by {@link Class#getSimpleName()}</dd>
390      * <dt>method-name</dt>
391      * <dd>as returned by {@link Method#getName()}</dd>
392      * <dt>class-simple-name</dt>
393      * <dd>hex string of whatever is returned by
394      * {@link Arrays#hashCode(Object[])}, wherein {@code Object[]} is the
395      * parameters of the test method</dd>
396      * <dt>thread-id</dt>
397      * <dd>identifier of thread as of {@link Thread#getId()}
398      * </dl>
399      * This means code running in {@link Test} annotated methods will have its
400      * thread named such this.
401      * </p>
402      * <p>
403      * Finally, notifies {@code Unitils} about being before method via
404      * {@link TestListener#beforeTestSetUp(Object, Method)}
405      * </p>
406      *
407      * @param testMethod
408      *            The test method, not null
409      * @param parameters
410      *            the parameters used to invoke this test method instance
411      */
412     @BeforeMethod(alwaysRun = true)
413     protected final void unitilsBeforeMethod(
414         final Method testMethod,
415         final Object[] parameters) {
416         currentThread().setName(SessionName.builder()
417             .className(testMethod.getDeclaringClass().getSimpleName())
418             .methodName(testMethod.getName())
419             .params(toHexString(Arrays.hashCode(parameters)))
420             .id(String.valueOf(currentThread().getId()))
421             .build()
422             .toString());
423 
424         log.info("starting test method {} with {} active threads",
425             currentThread().getName(),
426             concurrency.updateAndGet(
427                 previousCount -> max(previousCount, activeCount())));
428 
429         if (isBlank(randomId.get())) {
430             val temporaryRandomId = generateRandomId();
431             if (issuedRandomIds.add(temporaryRandomId)) {
432                 startTime.set(now());
433                 randomId.set(temporaryRandomId);
434                 log.debug("running on fresh thread, new random id {} set",
435                     randomId.get());
436             } else {
437                 throw new TestRuntimeException(
438                     "Internal error: duplicate random identifier found");
439             }
440         } else {
441             log.debug("running on previous thread, reusing random id {}",
442                 randomId.get());
443         }
444 
445         beforeTestSetUpCalled.set(Boolean.TRUE);
446 
447         // NOTE: should fail fast if not able to connect to database
448         // try {
449         getTestListener().beforeTestSetUp(this, testMethod);
450         // } catch (final Exception e) {
451         // log.warn("before test got {}", e.getMessage()); //$NON-NLS-1$
452         // }
453     }
