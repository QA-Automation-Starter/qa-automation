19  import java.lang.SuppressWarnings;
20  
21  import org.aspectj.lang.*;
22  import org.aspectj.lang.annotation.*;
23  
24  import com.tngtech.jgiven.report.model.*;
25  
26  import edu.umd.cs.findbugs.annotations.*;
27  
28  /**
29   * Patches {@link com.tngtech.jgiven.report.model.StepModel} in order to allow
30   * correct reporting of step execution duration.
31   * <p>
32   * https://github.com/TNG/JGiven/issues/755
33   * </p>
34   *
35   * <p>
36   * <strong>IMPORTANT:</strong> requires having
37   * {@code com.tngtech.jgiven:jgiven-core} as a weave dependency.
38   * </p>
39   */
40  @SuppressFBWarnings("MS_SHOULD_BE_FINAL")
41  @Aspect
42  public class StepModelPatchAspect {
43  
44      /**
45       * Monitors attempts to set step method's duration. If the duration is
46       * already set, then overrides to do nothing.
47       * 
48       * @param thisJoinPoint
49       *            this join-point
50       * @param stepModel
51       *            instance of {@link StepModel}
52       * @throws Throwable
53       *             any...
54       *
55       * @see #setDurationInNanos(StepModel)
56       */
57      // FIXME Join point return value will be lost
58      @Around(value = "setDurationInNanos(stepModel)",
59          argNames = "thisJoinPoint,stepModel") // for debugging info
60      @SuppressWarnings("static-method")
61      public void aroundSetDurationInNanos(
62          final ProceedingJoinPoint thisJoinPoint,
63          final StepModel stepModel) throws Throwable {
64          if (0 == stepModel.getDurationInNanos())
65              thisJoinPoint.proceed(); // as duration is not set, otherwise ovoid
66      }
67  
68      /**
69       * Matches the execution of {@link StepModel#setDurationInNanos(long)}
70       *
71       * @param stepModel
72       *            instance of {@link StepModel}
73       */
74      @Pointcut("execution("
75          + "void com.tngtech.jgiven.report.model.StepModel.setDurationInNanos(long))"
76          + "&& target(stepModel)")
77      public void setDurationInNanos(
78          @SuppressWarnings("unused") final StepModel stepModel) {
79          // nothing to do here -- just defines a pointcut matcher
80      }
81  }
