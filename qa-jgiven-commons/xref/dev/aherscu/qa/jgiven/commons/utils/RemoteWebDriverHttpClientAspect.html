39  /**
40   * Limits Selenium's remote low-level HTTP requests to configured by
41   * {@code poll.timeout} system property (in seconds). This is required in order
42   * to deal with remote requests hanging indefinitely.
43   */
44  @SuppressFBWarnings("MS_SHOULD_BE_FINAL")
45  @Aspect
46  @Slf4j
47  public class RemoteWebDriverHttpClientAspect implements ISuiteListener {
48      /**
49       * Instruments {@link HttpClient#execute(HttpRequest)}.
50       *
51       * @param thisJoinPoint
52       *            the execution
53       * 
54       * @return the {@link HttpResponse}
55       * @throws Throwable
56       *             any of below
57       * @throws TimeoutException
58       *             if the time limit is reached
59       * @throws InterruptedException
60       *             if the current thread was interrupted during execution
61       * @throws ExecutionException
62       *             if a checked exception was thrown
63       * @throws UncheckedExecutionException
64       *             if a {@code RuntimeException} was thrown
65       * @throws ExecutionError
66       *             if an {@code Error} was thrown
67       */
68      @Around("executeHttpRequestMethod()")
69      public Object aroundExecuteHttpRequestMethod(
70          final ProceedingJoinPoint thisJoinPoint)
71          throws Throwable {
72          // NOTE: HttpClient#execute(HttpRequest) has exactly one argument
73          val httpRequest = (HttpRequest) thisJoinPoint.getArgs()[0];
74          log.trace("executing {} {}",
75              httpRequest.getMethod(), httpRequest.getUri());
76          try {
77              val httpResponse =
78                  shouldExecuteWithoutTimeout(httpRequest)
79                      ? (HttpResponse) thisJoinPoint
80                          .proceed(thisJoinPoint.getArgs())
81                      : timeout(Unchecked
82                          .callable(() -> (HttpResponse) thisJoinPoint
83                              .proceed(thisJoinPoint.getArgs())),
84                          shouldExecuteFast(httpRequest)
85                              ? StageEx.pollTimeout.dividedBy(2)
86                              : StageEx.pollTimeout);
87              log.trace("executed {} {} -> {}",
88                  httpRequest.getMethod(), httpRequest.getUri(),
89                  httpResponse.getStatus());
90              return httpResponse;
91          } catch (final ExecutionException | UncheckedExecutionException e) {
92              log.error("failed due to ", e.getCause());
93              throw e.getCause();
94          } catch (final TimeoutException e) {
95              log.error("timed-out");
96              // NOTE possible error detection:
97              // send a status request with a few seconds timeout
98              // if it does not return throw a TestRuntimeException
99              // val instance = thisJoinPoint.getThis();
100             // val clientField = instance.getClass().getDeclaredField("client");
101             // val baseUrlField =
102             // instance.getClass().getDeclaredField("baseUrl");
103             // clientField.setAccessible(true);
104             // baseUrlField.setAccessible(true);
105             // val client = (okhttp3.OkHttpClient) clientField.get(instance);
106             // val baseUrl = (URL) baseUrlField.get(instance);
107             // val response = client.newCall(new Request.Builder()
108             // .url(UriBuilder.fromUri(baseUrl.toURI()).path("/status").build()
109             // .toURL())
110             // .build())
111             // .execute();
112             // val statusReady = new JSONObject(response.body().string())
113             // .getBoolean("ready");
114             throw e;
115         }
116     }

118     /**
119      * Marks the execution of {@link HttpClient#execute(HttpRequest)}, which is
120      * the low level SPI of Selenium's RemoteWebDriver HTTP connectivity.
121      */
122     @Pointcut("execution("
123         + "org.openqa.selenium.remote.http.HttpResponse "
124         + "org.openqa.selenium.remote.http.HttpClient+.execute("
125         + "org.openqa.selenium.remote.http.HttpRequest))")
126     public void executeHttpRequestMethod() {
127         // nothing to do here -- just defines a pointcut matcher
128     }
129 
130     @Override
131     public void onStart(final ISuite suite) {
132         log.trace("using time-out of {}", StageEx.pollTimeout);
133     }
134 
135     @Override
136     public void onFinish(final ISuite suite) {
137         log.trace("shutting down executor serivce");
138         ExecutorUtils.EXECUTOR_SERVICE.shutdownNow();
139     }
140 
141     private boolean shouldExecuteFast(final HttpRequest httpRequest) {
142         switch (httpRequest.getMethod()) {
143         case GET:
144             return true;
145         case DELETE:
146             return false;
147         case POST:
148             return httpRequest.getUri().endsWith("/element")
149                 || httpRequest.getUri().endsWith("/elements");
150         // TODO should somehow detect scrolling scripts
151         // || httpRequest.getUri().contains("/execute/")
152         // && httpRequest.get;
153         default:
154             throw new InvalidArgumentException(
155                 httpRequest.getMethod() + SPACE + httpRequest.getUri());
156         }
157     }
158 
159     private boolean shouldExecuteWithoutTimeout(final HttpRequest httpRequest) {
160         return "/session".equals(httpRequest.getUri());
161     }
