30  @SuppressFBWarnings("MS_SHOULD_BE_FINAL")
31  @Aspect
32  @Slf4j
33  public class DiagnosticsAspect {
34  
35      public static void selfTest() {
36          log.debug("self test");
37          throw new AssertionError("there was an error");
38      }
39  
40      @Around("selfTestExecution()")
41      @SuppressWarnings({ "static-method", })
42      public Object aroundSelfTest(
43          final ProceedingJoinPoint thisJoinPoint)
44          throws Throwable {
45          log.debug("before");
46          try {
47              log.debug("thisJoinPoint {}", thisJoinPoint);
48              log.debug("args {}", thisJoinPoint.getArgs());
49              return thisJoinPoint.proceed(thisJoinPoint.getArgs());
50          } finally {
51              log.debug("after");
52          }
53      }
54  
55      // see
56      // https://stackoverflow.com/questions/62250100/aspectj-execution-pointcut-on-same-method-throwing-npe
57      // https://stackoverflow.com/questions/58626598/aspectj-exceptions-handling-on-multiple-matching-advices
58      // https://bugs.eclipse.org/bugs/show_bug.cgi?id=552687
59      @Around("selfTestExecution2()")
60      @SuppressWarnings({ "static-method", })
61      public Object aroundSelfTest2(
62          final ProceedingJoinPoint thisJoinPoint)
63          throws Throwable {
64          log.debug("before2");
65          try {
66              log.debug("thisJoinPoint2 {}", thisJoinPoint);
67              log.debug("args2 {}", thisJoinPoint.getArgs());
68              return thisJoinPoint.proceed(thisJoinPoint.getArgs());
69          } finally {
70              log.debug("after2");
71          }
72      }
73  
74      @Pointcut("execution(void DiagnosticsAspect.selfTest())")
75      public void selfTestExecution() {
76          // nothing to do here -- just defines a pointcut matcher
77      }
78  
79      @Pointcut("execution(void DiagnosticsAspect.selfTest())")
80      public void selfTestExecution2() {
81          // nothing to do here -- just defines a pointcut matcher
82      }
83  }
