239     /**
240      * Attempts to initialize a Web driver with specified capabilities.
241      *
242      * <p>
243      * In addition, adds the Remote Web Driver session identifier to
244      * {@link #remoteSessions} mapped to name of test as defined by
245      * {@link ConfigurableScenarioTest#beforeClassInitializeSession} or
246      * {@link ConfigurableScenarioTest#beforeMethodInitializeSession}
247      * </p>
248      *
249      * @param capabilities
250      *            the capabilities; might be null; the capabilities may include
251      *            {@code name} for registering this web driver for some sort of
252      *            status reporting mechanism
253      * @return the initialized driver, or null if {@link DryRunAspect#dryRun}
254      *
255      * @throws RuntimeException
256      *             or derivative, if the initialization failed
257      */
258     public static WebDriverEx from(final Capabilities capabilities) {
259         log.debug("connecting session {} with {}",
260             capabilities.getCapability("sauce:name"), capabilities);
261 
262         return DryRunAspect.dryRun
263             ? null
264             : Failsafe.with(
265                 // NOTE: if an exception is not handled during @Before methods
266                 // JGiven report is not generated at all
267                 Fallback.of(e -> {
268                     log.debug("initialization failed due to {}",
269                         e.getLastFailure().getMessage());
270                     return null;
271                 }),
272                 new RetryPolicy<WebDriverEx>()
273                     // at most 3 retries by default
274                     .handle(WebDriverException.class)
275                     .onRetry(e -> log.debug("re-attempting connection #{}",
276                         e.getAttemptCount()))
277                     .onRetriesExceeded(e -> log.error("retries exceeded.",
278                         e.getFailure())))
279                 .onSuccess(e -> registerSession(e.getResult()))
280                 .get(() -> new WebDriverEx(webDriverFor(capabilities),
281                     capabilities));
282     }
283 
284     private static void registerSession(final WebDriverEx driver) {
285         val name = driver.originalCapabilities.getCapability("sauce:name");
286 
287         if (isNull(name)) {
288             log.warn("session name missing from capabilities"
289                 + " -- will not be reported");
290             return;
291         }
292 
293         val sessionNameWithTimestamp = SessionName.from(name.toString());
294         val sessionId = driver.asRemote().getSessionId();
295 
296         log.debug("registering remote web driver session id {} for {}",
297             sessionId.toString(), sessionNameWithTimestamp);
298 
299         log.debug("registered new session {}",
300             remoteSessions.put(sessionNameWithTimestamp,
301                 WebDriverSessionInfo.builder()
302                     .sessionId(sessionId)
303                     .capabilities(driver.originalCapabilities)
304                     .build()));
305     }

307     /**
308      * @return the registered Web Driver session informations for current test
309      *         method, including those registered during {@code @BeforeClass}
310      */
311     public static Iterable<WebDriverSessionInfo> sessionInfos() {
312         return Iterables.concat(
313             remoteSessions.get(SessionName
314                 .from(currentThread().getName())
315                 .asClassSession()),
316             remoteSessions.get(SessionName
317                 .from(currentThread().getName())));

320     @SneakyThrows
321     private static Class<? extends WebDriver> webDriverClassFor(
322         final Capabilities capabilities) {
323         val driverClass = Class.forName(requireNonNull(capabilities
324             .getCapability("-x:class"), "must have a class capability")
325             .toString())
326             .asSubclass(WebDriver.class);
327 
328         if (driverClass.isAssignableFrom(ChromeDriver.class))
329             WebDriverManager.chromedriver().setup();
330 
331         // ISSUE sometimes this fails with
332         // WebDriverManagerException: Error HTTP 403 executing
333         // https://api.github.com/repos/mozilla/geckodriver/releases
334         // TODO should find a method to retry these driver downloads
335         if (driverClass.isAssignableFrom(FirefoxDriver.class))
336             WebDriverManager.firefoxdriver().setup();
337 
338         if (driverClass.isAssignableFrom(EdgeDriver.class))
339             WebDriverManager.edgedriver().setup();
340 
341         if (driverClass.isAssignableFrom(SafariDriver.class))
342             WebDriverManager.safaridriver().setup();
343 
344         // if (driverClass.isAssignableFrom(OperaDriver.class))
345         // WebDriverManager.operadriver().setup();
346 
347         // NOTE there are drivers, like WinAppDriver, that are not supported
348         // by WebDriverManager -- in this case we just return the class
349         return driverClass;
350     }

352     /**
353      * Initializes a WebDriver for specified capabilities.
354      *
355      * @param capabilities
356      *            the capabilities
357      * @return the Web Driver
358      */
359     @SneakyThrows
360     public static WebDriver webDriverFor(final Capabilities capabilities) {
361         try {
362             return nonNull(capabilities.getCapability("-x:url"))
363                 ? webDriverClassFor(capabilities)
364                     .getConstructor(URL.class, Capabilities.class)
365                     .newInstance(new URL(
366                         requireNonNull(capabilities.getCapability("-x:url"),
367                             "INTERNAL ERROR: missing -x:url capability")
368                             .toString()),
369                         capabilities)
370                 : webDriverClassFor(capabilities)
371                     .getDeclaredConstructor()
372                     .newInstance();
373         } catch (final InvocationTargetException e) {
374             log.error("remote selenium connection failed due to:\n{}",
375                 e.getCause().getMessage());
376             throw e.getCause();
377         }
378     }

450      * @param event

453      *            the element

458     public boolean dispatch(final Event event, final WebElement element) {
459         return (Boolean) asJavaScriptExecutor()
460             .executeScript(
461                 "return arguments[0].dispatchEvent(" + event.toString() + ");",
462                 element);

516             .executeScript(
517                 "arguments[0].scrollIntoView();",
518                 element);
519         return element;
520     }

522     /**
523      * Retrieves Web context identifier of running mobile application.
524      *
525      * @return the identifier of first Web context
526      * @throws TestRuntimeException
527      *             if no Web context/view was found
528      */
529     @SneakyThrows
530     // ISSUE seems no longer in use -- maybe should be deprecated?
531     public String webContextIdentifier() {
532         // NOTE one iOS the initialization of web context is slow, hence this
533         return Failsafe.with(
534             Fallback.of(
535                 (CheckedConsumer<ExecutionAttemptedEvent<? extends String>>) e -> {
536                     throw new TestRuntimeException(e.getLastFailure());
537                 }),
538             new RetryPolicy<String>()
539                 // at most 3 retries by default
540                 .withMaxDuration(StageEx.pollTimeout)
541                 .handle(TestRuntimeException.class, WebDriverException.class)
542                 .onRetry(e -> log.debug("re-attempting finding web view #{}",
543                     e.getAttemptCount()))
544                 .onRetriesExceeded(e -> log.error("retries exceeded.",
545                     e.getFailure())))
546             .onSuccess(e -> log
547                 .debug("found web view context {}", e.getResult()))
548             .get(() -> ((ContextAware) driver)
549                 .getContextHandles()
550                 .stream()
551                 .peek(contextId -> log.trace("found context {}", contextId))
552                 .filter(contextId -> contextId.startsWith("WEBVIEW"))
553                 .findFirst()
554                 .orElseThrow(
555                     () -> new TestRuntimeException("no webview available")));
556     }
