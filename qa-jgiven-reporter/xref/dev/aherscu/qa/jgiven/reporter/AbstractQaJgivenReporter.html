38  /**
39   * Base functionality and defaults for all kinds of reporters.
40   * <p>
41   * Can be invoked as a TestNG Reporter, hence supports {@link IReporter} by
42   * implementing its {@link #generateReport(List, List, String)} and provides a
43   * no-args constructor. It also can be invoked from other workflow engine, such
44   * as from a Maven plugin, which should just call {@link #generate()} -- see the
45   * qa-testrail-reporter and qa-jgiven-reporter-maven-plugin sibling modules.
46   * </p>
47   * <p>
48   * Implementors are required to specify the {@link #generate()} method.
49   * </p>
50   *
51   * @see #with(XmlSuite) TestNG Reporter configuration
52   *
53   * @param <M>
54   *            one of JGiven's report models: {@link CompleteReportModel}, *
55   *            {@link ScenarioModel}, or {@link ReportModelFile}
56   * @param <T>
57   *            specific type of reporter
58   */
59  @SuperBuilder(toBuilder = true)
60  @NoArgsConstructor(force = true)
61  @Slf4j
62  @ToString
63  public abstract class AbstractQaJgivenReporter<M, T extends AbstractQaJgivenReporter<?, ?>>
64      implements IReporter {
65  
66      public static final String DEFAULT_REFERENCE_TAG    = "Reference";
67      public static final String DEFAULT_SCREENSHOT_SCALE = "0.2";
68      public static final String DEFAULT_DATE_PATTERN     = "yyyy-MMM-dd HH:mm O";
69  
70      @Builder.Default
71      protected final File       sourceDirectory          =
72          // FIXME Warning:(72, 40) 'Optional.get()' without 'isPresent()' check
73          Config.config().getReportDir().get();
74      @Builder.Default
75      protected final File       outputDirectory          =
76          new File(Config.config().getReportDir().get(), "qa-html");
77      @Builder.Default
78      protected final boolean    debug                    = false;
79      @Builder.Default
80      protected final String     screenshotScale          =
81          DEFAULT_SCREENSHOT_SCALE;
82      @Builder.Default
83      protected final String     datePattern              = DEFAULT_DATE_PATTERN;
84      @Builder.Default
85      protected final boolean    pdf                      = false;
86      @Builder.Default
87      protected final String     referenceTag             = DEFAULT_REFERENCE_TAG;
88      protected final String     templateResource;
89  
90      // see
91      // https://stackoverflow.com/questions/61633821/using-lombok-superbuilder-annotation-with-tobuilder-on-an-abstract-class
92      public abstract AbstractQaJgivenReporterBuilder<M, T, ?, ?> toBuilder();
93  
94      protected Mustache.Compiler compiler() {
95          return Mustache.compiler();
96      }
97  
98      protected QaJGivenReportModel<M> reportModel(final File targetReportFile) {
99          return QaJGivenReportModel.<M> builder()
100             .targetReportFile(targetReportFile)
101             .build();
102     }
103 
104     /**
105      * @param xmlSuites
106      *            The list of {@code XmlSuite}
107      * @param suites
108      *            The list of {@code ISuite}
109      * @param outputDirectory
110      *            The output directory is ignored, since it is specified by
111      *            JGiven reporter infrastructure
112      */
113     @Override
114     public void generateReport(
115         final List<XmlSuite> xmlSuites,
116         final List<ISuite> suites,
117         @SuppressWarnings("hiding") final String outputDirectory) {
118         xmlSuites.forEach(xmlSuite -> log.info("xml suite {}", xmlSuite));
119         // ISSUE: should be empty for xml driven invocations (?)
120         // if yes, then should throw an unsupported exception
121         suites.forEach(suite -> log.info("suite {}", suite.getName()));
122 
123         xmlSuites.forEach(xmlSuite -> with(xmlSuite).prepare().generate());
124     }
125 
126     /**
127      * Builds a new reporter configured per following TestNG XML suite
128      * parameters:
129      * <dl>
130      * <dt>referenceTag</dt>
131      * <dd>the reference tag identifier, or {@link #DEFAULT_REFERENCE_TAG}</dd>
132      * <dt>screenshotScale</dt>
133      * <dd>the screenshot scale to apply when embedding files into reports, or
134      * {@link #DEFAULT_SCREENSHOT_SCALE}</dd>
135      * <dt>datePattern</dt>
136      * <dd>the date pattern to use for presenting dates, or
137      * {@link #DEFAULT_DATE_PATTERN}</dd>
138      * <dt>templateResourceXXX</dt>
139      * <dd>the template resource file name to apply; the {@code XXX} is the
140      * concrete reporter implementation name (class)</dd>
141      * </dl>
142      *
143      * @param xmlSuite
144      *            TestNG XML suite
145      * @return reporter configured
146      */
147     protected AbstractQaJgivenReporter<M, T> with(final XmlSuite xmlSuite) {
148         return this
149             // NOTE see
150             // https://stackoverflow.com/questions/56761054/lombok-wither-with-inheritance-super-sub-classes
151             .toBuilder()
152             .referenceTag(defaultIfBlank(xmlSuite.getParameter("referenceTag"),
153                 referenceTag))
154             .screenshotScale(defaultIfBlank(
155                 xmlSuite.getParameter("screenshotScale"), screenshotScale))
156             .datePattern(defaultIfBlank(xmlSuite.getParameter("datePattern"),
157                 datePattern))
158             .templateResource(templateResourceParamFrom(xmlSuite,
159                 templateResource))
160             .build();
161     }
162 
163     @SneakyThrows
164     public AbstractQaJgivenReporter<M, T> prepare() {
165         log.info("configuration {}", this);
166 
167         forceMkdir(outputDirectory);
168 
169         return this;
170     }
171 
172     protected String templateResourceParamFrom(
173         final XmlSuite xmlSuite,
174         final String defaultTemplateResource) {
175         return defaultIfBlank(
176             xmlSuite.getParameter(
177                 "templateResource" + this.getClass().getSimpleName()),
178             defaultTemplateResource);
179     }
180 
181     abstract public void generate();
182 
183     protected File reportFile(final File reportModelFile,
184         final String extension) {
185         return new File(outputDirectory, reportModelFile.getName() + extension);
186     }
187 
188     protected Collection<File> listJGivenReports() {
189         return listFiles(sourceDirectory, new SuffixFileFilter(".json"), null);
190     }
191 
192     protected Template template() {
193         return TemplateUtils.using(compiler()).loadFrom(templateResource);
194     }
195 }
