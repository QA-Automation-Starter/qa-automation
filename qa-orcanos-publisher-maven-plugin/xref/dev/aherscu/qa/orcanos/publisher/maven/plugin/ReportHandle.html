33  /**
34   * Encapsulates a report file with its status and attributes to be reported to
35   * Orcanos.
36   */
37  @SuppressFBWarnings(value = "EI_EXPOSE_REP2", justification = "value object")
38  @Builder
39  @ToString
40  final class ReportHandle {
41  
42      static final String               DEVICE_NAME      = "DeviceName";
43      static final String               FAIL             = "FAIL";
44      static final String               PASS             = "PASS";
45      static final String               PLATFORM_NAME    = "PlatformName";
46      static final String               PLATFORM_VERSION = "PlatformVersion";
47  
48      // NOTE: this is the defined in qa-jgiven-commons as an annotation;
49      // we do not include that dependency here in order to make the build faster.
50      static final String               REFERENCE        =
51          "dev.aherscu.qa.jgiven.commons.tags.Reference";
52      static final String               TAG_SEPARATOR    = COLON;
53      private final File                sourceFile;
54      private final String              status;
55      private final String              tag;
56      private final Map<String, String> attributes;
57  
58      /**
59       * Returns the device name from {@code DeviceName} attribute.
60       * 
61       * @return the device name
62       * 
63       */
64      public String deviceName() {
65          return attributes.get(DEVICE_NAME);
66      }
67  
68      /**
69       * Returns the execution set identifier from {@code Reference} attribute.
70       * 
71       * @return the id
72       * 
73       */
74      public String executionSetId() {
75          return substringAfter(reference(), UNDERSCORE);
76      }
77  
78      /**
79       * Returns supported {@code Reference} has an execution set id and a test
80       * case id.
81       * 
82       * @return true if has a execution set id
83       * 
84       */
85      public boolean hasSupportedReference() {
86          return reference().matches("\\d+_\\d+");
87      }
88  
89      /**
90       * Returns supported status can be {@code SUCCESS</tt> or <tt>FAIL}.
91       * 
92       * @return true if supported
93       * 
94       */
95      public boolean hasSupportedStatus() {
96          return !Status.UNSUPPORTED.equals(status());
97      }
98  
99      /**
100      * Returns the platform name from {@code PlatformName} attribute.
101      * 
102      * @return platform name
103      * 
104      */
105     public String platformName() {
106         return attributes.get(PLATFORM_NAME);
107     }
108 
109     /**
110      * Returns the platform version from {@code PlatformVersion} attribute.
111      * 
112      * @return platform version
113      * 
114      */
115     public String platformVersion() {
116         return attributes.get(PLATFORM_VERSION);
117     }
118 
119     public String reference() {
120         Iterable<String> references = Splitter.onPattern(COMMA)
121             .split(defaultIfNull(attributes.get(REFERENCE), EMPTY));
122 
123         var defaultReference = EMPTY;
124         for (val r : references) {
125             val taggedReference = r.contains(TAG_SEPARATOR)
126                 ? Pair.of(
127                     substringBefore(r, TAG_SEPARATOR),
128                     substringAfter(r, TAG_SEPARATOR))
129                 : Pair.of(EMPTY, r);
130             if (defaultReference.isEmpty()
131                 && taggedReference.getKey().equals(EMPTY))
132                 defaultReference = taggedReference.getValue();
133             if (taggedReference.getKey().equals(tag))
134                 return taggedReference.getValue();
135         }
136         return defaultReference;
137     }
138 
139     /**
140      * Returns the report source file to upload.
141      * 
142      * @return the source file
143      * 
144      */
145     public File sourceFile() {
146         return sourceFile;
147     }
148 
149     /**
150      * Returns the execution status.
151      * 
152      * @return the status
153      * 
154      */
155     public Status status() {
156         return Status.from(status);
157     }
158 
159     /**
160      * Returns the test case identifier from {@code Reference} attribute.
161      * 
162      * @return the test id
163      */
164     public String testId() {
165         return substringBefore(reference(), UNDERSCORE);
166     }
167 
168     @SuppressWarnings("hiding")
169     enum Status {
170         PASS, FAIL, UNSUPPORTED;
171 
172         public static final String FAILED  = "FAILED";
173         public static final String SUCCESS = "SUCCESS";
174 
175         static Status from(final String value) {
176             switch (String.valueOf(value)) {
177             case SUCCESS:
178                 return PASS;
179             case FAILED:
180                 return FAIL;
181             default:
182                 return UNSUPPORTED;
183             }
184         }
185     }
186 }
