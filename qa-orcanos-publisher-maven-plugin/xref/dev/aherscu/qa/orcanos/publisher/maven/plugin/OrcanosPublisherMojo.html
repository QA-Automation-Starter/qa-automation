54  /**
55   * Orcanos {@code publish} goal should be configured as in
56   * {@code plugin-config.xml} file used for testing.
57   */
58  @SuppressWarnings("ClassWithTooManyFields")
59  @Mojo(name = "publish",
60      threadSafe = true,
61      defaultPhase = LifecyclePhase.POST_INTEGRATION_TEST)
62  public class OrcanosPublisherMojo extends AbstractMojo {
63      static final String GET_EXECUTION_RUN_DETAILS =
64          "Get_Execution_Run_Details_XML";
65      static final String PROJECT_ID                = "ProjectID";
66      static final String RECORD_EXECUTION_RESULT   =
67          "Record_Execution_Results_New";
68      static final String UPLOAD_ATTACHMENT         = "QW_Add_MiscAttachment";
69  
70      static {
71          // NOTE: enables SSL without root validation
72          TrustAllX509TrustManager.disableSslCertificateValidation();
73      }
74  
75      @Parameter(property = "orcanos.skip", defaultValue = "false")
76      protected boolean    skip;
77      @Parameter(
78          defaultValue = "${project.build.directory}/jgiven-reports/qa-html")
79      protected File       reportsDirectory;
80      /**
81       * Groups to match in report file to be published as follows:
82       * <ol>
83       * <li>absolute path to file</li>
84       * <li>execution status</li>
85       * <li>class name</li>
86       * <li>method name</li>
87       * </ol>
88       * Currently, only the execution status is used.
89       */
90      @Parameter(
91          defaultValue = "(.*[\\\\|/])([^\\\\|/]+)-([^\\\\|/]+)-([^\\\\|/]+)$")
92      protected String     filePattern;
93      /**
94       * Files having this extension will be scanned and uploaded if matched
95       * correctly by {@link #filePattern}.
96       */
97      @Parameter(defaultValue = ".html")
98      protected String     reportFileExtension;
99      @Parameter(required = true)
100     protected URI        orcanosUrl;
101     @Parameter(required = true)
102     protected String     orcanosProjectId;
103     @Parameter(defaultValue = "10000")
104     protected int        connectTimeoutMs;
105     @Parameter(defaultValue = "10000")
106     protected int        readTimeoutMs;
107     @Parameter(defaultValue = "3")
108     protected int        retriesLimit;
109     @Parameter
110     protected String     tag;
111     @Parameter
112     protected Properties additionalExecutionFields;
113 
114     @Override
115     public void execute() throws MojoExecutionException {
116         if (skip) {
117             getLog().info("skipping");
118             return;
119         }
120 
121         getLog().info("publishing to Orcanos from: " + reportsDirectory);
122         getLog().info("using file pattern: " + filePattern);
123         getLog().info("for files ending with: " + reportFileExtension);
124         getLog().info("Orcanos endpoint URL: " + orcanosUrl);
125         getLog().info("Orcanos Project ID: " + orcanosProjectId);
126         getLog().info("Connection timeout (ms): " + connectTimeoutMs);
127         getLog().info("Read timeout (ms): " + readTimeoutMs);
128         getLog().info("Retries limit: " + retriesLimit);
129         getLog().info("Tag: " + tag);
130         getLog().info("Additional execution fields: "
131             + additionalExecutionFields);
132 
133         val compiledReferencePattern = compile(filePattern);
134 
135         try (val client = orcanosClient()) {
136             listReportFiles()
137                 // ISSUE cannot parallelize because Orcanos reports same run;
138                 // the "runs" are identified with second granularity timestamp
139                 .stream()
140                 .peek(file -> getLog().info("found report: " + file.getName()))
141                 .map(file -> compiledReferencePattern
142                     .matcher(file.getAbsolutePath()))
143                 .filter(Matcher::find)
144                 .map(Unchecked.function(matcher -> ReportHandle.builder()
145                     .sourceFile(new File(matcher.group(0)))
146                     .status(matcher.group(2))
147                     .attributes(readAttributesOf(new File(matcher.group(0))))
148                     .tag(tag)
149                     .build()))
150                 .peek(reportHandle -> getLog()
151                     .info("parsed meta-info: " + reportHandle.toString()))
152                 .filter(ReportHandle::hasSupportedStatus)
153                 .filter(ReportHandle::hasSupportedReference)
154                 .peek(reportHandle -> getLog().info("meta-info supported"))
155                 .forEach(reportHandle -> report(client.target(orcanosUrl),
156                     reportHandle));
157         } catch (final Throwable t) {
158             throw new MojoExecutionException("configuration error", t);
159         }
160     }
161 
162     private Collection<File> listReportFiles() {
163         val reportFiles = listFiles(reportsDirectory,
164             new SuffixFileFilter(reportFileExtension),
165             null);
166 
167         if (reportFiles.isEmpty())
168             throw new RuntimeException("no reports to upload");
169 
170         return reportFiles;
171     }
172 
173     private AutoCloseableClient orcanosClient() {
174         return new AutoCloseableClient(LoggingClientBuilder
175             .newClient()
176             .register(RxFlowableInvokerProvider.class)
177             .register(MultiPartFeature.class)
178             .register(HttpAuthenticationFeature.basicBuilder()
179                 .credentials(
180                     usernameFrom(orcanosUrl),
181                     passwordFrom(orcanosUrl))
182                 .build())
183             .property(CONNECT_TIMEOUT, connectTimeoutMs)
184             .property(READ_TIMEOUT, readTimeoutMs));
185     }
186 
187     private GenericResponse recordExecutionResults(
188         final WebTarget orcanosEndpointTarget,
189         final ReportHandle reportHandle,
190         final ExecutionSetRunResultsEx results,
191         final GenericResponse uploadConfirmation) {
192         getLog().info(format("reporting run {0} of test case {1} "
193             + "on execution set {2} with status {3}",
194             requireNonNull(results.getRun(),
195                 "couldn't get execution run result, check orcanos credentials")
196                     .getName(),
197             reportHandle.testId(),
198             reportHandle.executionSetId(),
199             reportHandle.status()));
200 
201         return orcanosEndpointTarget
202             .path(RECORD_EXECUTION_RESULT)
203             .request(APPLICATION_JSON)
204             .rx(RxFlowableInvoker.class)
205             .post(json(RecordExecutionResults.builder()
206                 .executionSetRunResults(results
207                     .withAdditionalFields(additionalExecutionFields)
208                     .withDeviceInfo(
209                         reportHandle.deviceName(),
210                         reportHandle.platformName(),
211                         reportHandle.platformVersion())
212                     .withStatus(reportHandle.status().name())
213                     .withAttachment("click to open automation report",
214                         uploadConfirmation.href()))
215                 .build()),
216                 GenericResponse.class)
217             .retry(retriesLimit, e -> e instanceof ProcessingException)
218             .blockingFirst();
219     }
220 
221     private void report(
222         final WebTarget orcanosEndpointTarget,
223         final ReportHandle reportHandle) {
224 
225         val attachmentId = randomUUID();
226         try (val formDataMultiPart = new FormDataMultiPart();
227             val inputStream = openInputStream(reportHandle.sourceFile());
228             val bufferedInputStream = buffer(inputStream);
229             val reportToUpload =
230                 formDataMultiPart.bodyPart(new StreamDataBodyPart(
231                     attachmentId.toString(),
232                     bufferedInputStream,
233                     attachmentId + reportFileExtension,
234                     TEXT_HTML_TYPE))) {
235 
236             getLog().info(
237                 recordExecutionResults(
238                     orcanosEndpointTarget,
239                     reportHandle,
240                     retrieveExecutionSetRunResults(orcanosEndpointTarget,
241                         reportHandle),
242                     uploadReport(orcanosEndpointTarget, reportToUpload))
243                         .toString());
244         } catch (final Throwable t) {
245             getLog().error(reportHandle.toString(), t);
246         }
247     }
248 
249     private ExecutionSetRunResultsEx retrieveExecutionSetRunResults(
250         final WebTarget orcanosEndpointTarget,
251         final ReportHandle reportHandle) {
252         return orcanosEndpointTarget
253             .register(ExecutionSetRunResultsProvider.class)
254             // ISSUE if the credentials are wrong
255             // the requests still returns with 200 (success)
256             // but with an "error" response body like this
257             // <Error xmlns="">
258             // <ErrorStatus>1</ErrorStatus>
259             // <ErrorInfo>Email / User Name or password are
260             // incorrect</ErrorInfo>
261             // <ErrorTrace>Initialize_new</ErrorTrace>
262             // </Error>
263             // Perhaps the best way to deal with this is to change
264             // the Execution_Set_Run_Results schema in order to allow such
265             // alternative structure. It will still require a verification,
266             // but will allow access to error message.
267             // Meanwhile, we will just throw an NPE with a generic message,
268             // see recordExecutionResults above.
269             .path(GET_EXECUTION_RUN_DETAILS)
270             .request(APPLICATION_XML)
271             .rx(RxFlowableInvoker.class)
272             .post(json(ExecutionSet.builder()
273                 .testId(reportHandle.testId())
274                 .executionSetId(reportHandle.executionSetId())
275                 .build()),
276                 ExecutionSetRunResultsEx.class)
277             .retry(retriesLimit, e -> e instanceof ProcessingException)
278             .blockingFirst();
279     }
280 
281     private GenericResponse uploadReport(
282         final WebTarget orcanosEndpointTarget,
283         final MultiPart attachmentMultiPart) {
284         return orcanosEndpointTarget
285             .path(UPLOAD_ATTACHMENT)
286             .queryParam(PROJECT_ID, orcanosProjectId)
287             .request(APPLICATION_JSON)
288             .rx(RxFlowableInvoker.class)
289             .post(entity(attachmentMultiPart,
290                 attachmentMultiPart.getMediaType()),
291                 GenericResponse.class)
292             .retry(retriesLimit, e -> e instanceof ProcessingException)
293             .blockingFirst();
294     }

298         // nothing special
