40  /**
41   * Generic REST client actions. Act on previously set client. Each action closes
42   * the connection.
43   *
44   * @param <SELF>
45   *            the type of the subclass
46   *
47   * @author aherscu
48   */
49  @Slf4j
50  @ThreadSafe
51  public class RestActions<SELF extends RestActions<SELF>>
52      extends GenericActions<RestScenarioType, SELF> {
53      /**
54       * Target ({@link WebTarget}) of Web actions in this class.
55       */
56      protected final ThreadLocal<WebTarget> target          =
57          new ThreadLocal<>();
58      /**
59       * The retrieved {@link Response}'s contents. We hold this as a string
60       * because reading the contents of a {@link Response} cannot be done twice.
61       */
62      @ProvidedScenarioState
63      protected final ThreadLocal<String>    responseContent =
64          new ThreadLocal<>();
65      /**
66       * FIXME: this is a patch. We should hold an extended version of Response
67       * that holds the contents as a string (see above responseContent field).
68       * Perhaps this should be done in the AutoCloseableResponse class. This has
69       * to be done in a separate version of this library in order no to break
70       * older clients.
71       * <p>
72       * The retrieved {@link Response} in closed state.
73       */
74      @ProvidedScenarioState
75      protected final ThreadLocal<Response>  closedResponse  =
76          new ThreadLocal<>();
77      /**
78       * The given {@link Client}. This client is released upon scenario
79       * completion.
80       *
81       * @see RestVerifications
82       */
83      @ExpectedScenarioState
84      protected ThreadLocal<Client>          client;

102     /**
103      * Creates a {@link WebTarget} object.
104      *
105      * @param url
106      *            the URL
107      * @throws NullPointerException
108      *             in case the supplied argument is null
109      * @throws IllegalArgumentException
110      *             in case the supplied string is not a valid URI template
111      * @return {@link #self()}
112      */
113     public SELF connecting_to(final @NotNull URI url) {
114         log.debug("connecting to {}", url);
115         target.set(requireNonNull(client.get(), "must set a REST client before")
116             .target(url));
117         return self();
118     }

245     /**
246      * Fills a value into a query parameter.
247      *
248      * @param value
249      *            the value; if the value is an {@link Iterable}, then its items
250      *            are concatenated with comma delimiters
251      * @param name
252      *            the parameter's name
253      * @throws IllegalStateException
254      *             if {@link #connecting_to(URI)} has not been called before
255      * @return {@link #self()}
256      */
257     public SELF setting_$_into_query_param(
258         final Object value,
259         final String name) {
260         target.set(target.get()
261             .queryParam(name,
262                 value instanceof Iterable
263                     ? join((Iterable<?>) value, COMMA)
264                     : value));
265         return self();
266     }

268     /**
269      * Invokes specified REST operation.
270      *
271      * @param operation
272      *            the REST operation to invoke
273      * @throws IllegalStateException
274      *             if {@link #connecting_to(URI)} has not been called before
275      * @return {@link #self()}
276      */
277     protected final SELF invoke(
278         final Function<Invocation.Builder, Response> operation) {
279         return Failsafe.with(retryPolicy)
280             .get(() -> {
281                 try (val response = new AutoCloseableResponse(operation
282                     .apply(target.get().request()))) {
283                     closedResponse.set(response);
284                     log.trace(">>> stored closed response into {}:{}",
285                         closedResponse, this);
286                     responseContent.set(response.readEntity(String.class));
287                     log.trace(">>> stored content into {}:{}",
288                         responseContent, this);
289                 }
290                 // ISSUE on jdk11+ fails to compile via maven due to missing
291                 // (SELF) cast
292                 return self();
293             });
294     }

296     /**
297      * Creates a step function which when called, executes specified invocation,
298      * stores the closed response and response content, and calls the specified
299      * consumer on response content.
300      * <p>
301      * Attaches the response.
302      *
303      * @param invocation
304      *            a JAX-RS invocation
305      * @param consumer
306      *            something to execute on response content
307      * @return step function that executes the specified invocation
308      */
309     protected final Function<SELF, SELF> invoke(
310         final Invocation invocation,
311         final Consumer<String> consumer) {
312         return self -> {
313             try (val response = invocation.invoke()) {
314                 log.trace("invoking {}", invocation);
315                 closedResponse.set(response);
316                 responseContent.set(attach(response.readEntity(String.class)));
317                 consumer.accept(responseContent.get());
318                 return self;
319             }
320         };
321     }
322 
323     /**
324      * Creates a step function which when called, executes specified invocation,
325      * stores the closed response and response content.
326      * <p>
327      * Attaches the response.
328      *
329      * @param invocation
330      *            a JAX-RS invocation
331      * @return step function that executes the specified invocation
332      */
333     protected final Function<SELF, SELF> invoke(
334         final Invocation invocation) {
335         return invoke(invocation, identity()::apply);
336     }
